int Compte(int cat)
{
	int		i, total;
	object	p;

	total = 0;
	for ( i=0 ; i<10000 ; i++ )
	{
		p = retobject(i);
		if ( p == null )  break;
		if ( p.category == cat )  total ++;
	}
	return total;
}
