extern void object::Activity( )
{
	errmode(0);  // ne stoppe pas si erreur

	while ( true )
	{
		motor(-0.2, 0.2);
		wait(1);
		motor(0.2, -0.2);
		wait(1);
		motor(0, 0);
		wait(4);

		motor(0.4, -0.4);
		wait(2);
		motor(0, 0);
		wait(1);
		motor(-0.4, 0.4);
		wait(2);
		motor(0, 0);
		wait(1);
	}
}
