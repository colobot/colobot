// crée un objet et retourne son pointeur
CPoint newcpoint()
{
	CPoint p = new CPoint ( 3, 3 );
	return p;
}

extern public void y ()
{
	CPoint test = newcpoint();
	println ( test );
	dontmodif( test );
	println ( test );
}

// ne doit pas modifier l'objet en paramètre
void dontmodif ( CPoint pp )
{
	pp.x = 5;
	pp.y = 2;
	println ( pp, pp.x, pp.y );
}
